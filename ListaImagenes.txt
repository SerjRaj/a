3. Crear una clase java llamada 'Adaptador' (esta debe incluir un :

    Context context;
    String lista1[];
    String lista2[];
    int listaimagenes[];
    LayoutInflater inflater;


    public Adaptador(Context context, String[] lista1, String[] lista2, int[] listaimagenes) {
        this.context = context;
        this.lista1 = lista1;
        this.lista2 = lista2;
        this.listaimagenes = listaimagenes;
        this.inflater = LayoutInflater.from(context);
    }

    @Override
    public int getCount() {

        return (lista2 != null) ? lista2.length : 0;
    }

    @Override
    public Object getItem(int position) {
        return lista1[position];
    }

    @Override
    public long getItemId(int position) {
        return position;
    }

    @Override
    public View getView(int position, View convertView, ViewGroup parent) {
        if (convertView == null) {
            convertView = inflater.inflate(R.layout.basurita, parent, false);
        }

        TextView textview = convertView.findViewById(R.id.textview);
        TextView textview2 = convertView.findViewById(R.id.textview2);
        ImageView img = convertView.findViewById(R.id.imageicon);

        textview.setText(lista1[position]);
        textview2.setText(lista2[position]);
        deporteimg.setImageResource(listaimagenes[position]);

        return convertView;
    }
