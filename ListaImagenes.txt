3. Crear una clase java llamada 'Adaptador' (esta debe incluir un :

   
    Context contexto;
    String[] arreglo1;
    String[] arreglo2;
    int[] imagenes;
    LayoutInflater inflater;

    public Adaptador(Context contexto, String[] arreglo1, String[] arreglo2, int[] imagenes) {
        this.contexto = contexto;
        this.arreglo1 = arreglo1;
        this.arreglo2 = arreglo2;
        this.imagenes = imagenes;
        this.inflater = LayoutInflater.from(contexto);
    }

    @Override
    public int getCount() {
        return arreglo1.length;
    }

    @Override
    public Object getItem(int position) {
        return arreglo1[position];
    }

    @Override
    public long getItemId(int position) {
        return position;
    }

    @Override
    public int getViewTypeCount() {
        return 1;
    }

    @Override
    public int getItemViewType(int position) {
        return 0;
    }

    @Override
    public View getView(int position, View convertView, ViewGroup parent) {

        View v = convertView;
        if (v == null) {
            v = inflater.inflate(R.layout.basurita, parent, false);
        }

        TextView t1 = v.findViewById(R.id.textview);
        TextView t2 = v.findViewById(R.id.textview2);
        ImageView iv = v.findViewById(R.id.imageicon);

        t1.setText(arreglo1[position]);
        t2.setText(arreglo2[position]);
        iv.setImageResource(imagenes[position]);

        return v;
    }
